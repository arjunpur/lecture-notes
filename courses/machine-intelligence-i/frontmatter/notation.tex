\section*{Learning and optimization}

\begin{tabular}{@{}ll@{}}
  \toprule
  \textbf{Symbol} & \textbf{Meaning} \\
  \midrule
  $\vec{x}$ & input vector \\
  $y$ & target or output \\
  $\wvec$ & parameter vector or weight vector \\
  $\ell(\wvec;\vec{x},y)$ & loss on one sample \\
  $R(\wvec)$ & true risk \\
  $R_{\mathrm{emp}}(\wvec)$ & empirical risk \\
  \bottomrule
\end{tabular}

\section*{Neural networks}

\begin{tabular}{@{}ll@{}}
  \toprule
  \textbf{Symbol} & \textbf{Meaning} \\
  \midrule
  $a_j$ & pre-activation \\
  $h_j$ & hidden activation \\
  $\sigma(\cdot)$ & elementwise nonlinearity \\
  $\Wmat$ & weight matrix \\
  $\delta_j$ & backpropagated error signal \\
  \bottomrule
\end{tabular}

\section*{Probability and reinforcement learning}

\begin{tabular}{@{}ll@{}}
  \toprule
  \textbf{Symbol} & \textbf{Meaning} \\
  \midrule
  $p(\cdot)$ & probability density or mass function \\
  $\E[\cdot]$ & expectation \\
  $s_t, a_t, r_t$ & state, action, reward at time $t$ \\
  $V^\pi(s)$ & value of state $s$ under policy $\pi$ \\
  $Q^\pi(s,a)$ & action value under policy $\pi$ \\
  $\gamma$ & discount factor \\
  \bottomrule
\end{tabular}
