\section*{Spike-train notation}

\begin{tabular}{@{}ll@{}}
  \toprule
  \textbf{Symbol} & \textbf{Meaning} \\
  \midrule
  $t_i$ & Time of spike $i$ \\
  $\rho(t)$ & Spike-train response function \\
  $r(t)$ & Estimated firing-rate curve \\
  $\Delta t$ & Temporal bin width or smoothing scale \\
  \bottomrule
\end{tabular}

\section*{Probability and statistics}

\begin{tabular}{@{}ll@{}}
  \toprule
  \textbf{Symbol} & \textbf{Meaning} \\
  \midrule
  $p(x)$ & Probability density or mass function \\
  $\E[\cdot]$ & Expectation \\
  $\Var[\cdot]$ & Variance \\
  $C_{xy}(\tau)$ & Cross-covariance at lag $\tau$ \\
  \bottomrule
\end{tabular}

\section*{Linear algebra}

\begin{tabular}{@{}ll@{}}
